<script>
// --- Router très simple (hash-based) ---
const routes = {
'#/': 'home', '#/manifeste': 'manifeste', '#/annexes': 'annexes',
'#/atelier': 'atelier', '#/historique': 'historique', '#/licence': 'licence'
};
const sections = Object.values(routes).map(id => document.getElementById(id));
function render() {
const hash = location.hash || '#/';
sections.forEach(s => s.classList.remove('active'));
const id = routes[hash] || 'home';
document.getElementById(id).classList.add('active');
document.querySelectorAll('[data-route]').forEach(a => a.setAttribute('aria-current', a.getAttribute('href')===hash ? 'page':'false'));

}
addEventListener('hashchange', render); render();

// --- Mode fragmentaire ---
const fragToggle = document.getElementById('fragmentToggle');
const fragParas = [...document.querySelectorAll('[data-frag]')];
fragToggle.addEventListener('change', () => {
if (!fragToggle.checked) { fragParas.forEach(p => p.style.display=''); return; }
// Affiche aléatoirement ~50% des fragments
fragParas.forEach(p => p.style.display = Math.random() < 0.5 ? 'block' : 'none');
});

// --- Partage lien section ---
const shareBtn = document.getElementById('shareBtn');
const shareState = document.getElementById('shareState');
shareBtn.addEventListener('click', async () => {
try {
await navigator.clipboard.writeText(location.href);
shareState.textContent = 'Lien copié.';
setTimeout(()=>shareState.textContent='', 2000);
} catch (e) {
shareState.textContent = 'Copie impossible.';
}
});

// --- Atelier: générateur de fragments ---
const FRAG_POOL = [
'Regarder sans capturer.',
'Nommer, puis effacer la trace du nom.',
'Une identité est un outil, pas un sanctuaire.',
'La fiction est une méthode opératoire.',
'Se soustraire n\'est pas se taire.',
'Éviter les miroirs totalisants.',
'Conserver une marge d\'ombre utile.',
'Parler juste assez pour déplacer.',
'Changer de point d\'appui, pas de monde.',
'Le dossier n\'est pas la personne.'
];
const genBtn = document.getElementById('genFrag');
const outFrag = document.getElementById('outFrag');
genBtn.addEventListener('click', () => {
const picks = [...FRAG_POOL].sort(()=>Math.random()-0.5).slice(0,5);
outFrag.textContent = picks.map(x=>'• '+x).join('\n');
});

// --- Historique minimal (local, anonyme) ---
const log = document.getElementById('log');
const entries = [
['2025-08-12','Initialisation du prototype statique.'],
['2025-08-12','Activation du mode fragmentaire et de l\'atelier local.']
];
entries.forEach(([d,t])=>{
const li=document.createElement('li'); li.textContent=`${d} — ${t}`; log.appendChild(li);
});
</script>
